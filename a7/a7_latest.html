filename <- "State_to_State_Migrations_Table_2018.xlsm"
tbl = tidyxl::xlsx_cells(filename)
#head(tbl)
data_fileter <- tbl %>% filter(row >= 7) %>% 
  behead('W', source) %>% behead('N', target) %>% 
  select(source, target, numeric) %>% 
  filter(source %in% c('New York', 'Texas', 'Virginia', 'Michigan')) %>% 
  filter(!target %in% c('Total', 'Puerto Rico', 'U.S. Island Area', 'Foreign Country')) %>%     drop_na() %>%  rename(count = numeric)  %>% group_by(source) %>% 
   slice(seq_len(10))
library(igraph)
